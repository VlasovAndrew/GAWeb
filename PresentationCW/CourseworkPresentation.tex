\begin{frame}
    \frametitle{Цели и задачи \slnumber}
    \textbf{Цель работы:} создать приложение для исследования параметров генетического алгоритма для поиска центральных вершин.
    
    \textbf{Задачи:} 
    \begin{itemize}
    	\item разработать способ загрузки и хранения графов,
    	\item создать интерфейс, позволяющий пользователю запускать алгоритм,
    	\item создать гибкое программное решение, которое можно было бы легко изменять и использовать.
    \end{itemize}
\end{frame}
